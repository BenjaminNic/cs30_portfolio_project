def list_permutations(words)
  a = []
  a << words
  b = permutations(a)
  c = remove_dup(b)
  p c
end
# a = ["word"]
def permutations(word)
  if word[0].size == 1
    return word
  else
    word.each do |i|
      b = i.split(//)
      c = b.shift
      y = [] << b.join
      return place(c, permutations(y))
    end
  end
end
# b = w, word = ["ord", "odr", "rod", "rdo", "dor", "dro"]
def place(x, y)
  holder = []
  y.each do |i|
    k = i.size
    j = 0
    0.upto(k) do |a|
      holder << i.dup.insert(j, x)
      j += 1
    end
  end
  return holder
end
def remove_dup(x)
  a = []
  x.each do |i|
    if a.count(i) == 0
      a << i
    else
      next
    end
  end
  return a
end
